**Python中的枚举类**
```python
from enum import Enum

# 定义枚举类
class Color(Enum):
    RED = 1
    GREEN = 2
    BLUE = 3

# 使用枚举成员
print(Color.RED)  # 输出: Color.RED
print(Color(2))   # 输出: Color.GREEN
print(Color.GREEN.name)  # 输出: 'GREEN'
print(Color.BLUE.value)  # 输出: 3

# 枚举成员迭代
for color in Color:
    print(color)

# 比较枚举成员
print(Color.RED == Color.RED)  # 输出: True
print(Color.RED is Color.GREEN)  # 输出: False
```

class Config:
    @property
    def PI(self):
        return 3.14159

config = Config()
print(config.PI)  # ✅ 输出: 3.14159
config.PI = 3.14  # ❌ AttributeError: can't set attribute
